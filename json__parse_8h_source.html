161int json_marshal_array_indent_int(int* obj, int len, int indent, int curindent,
162 sstr_t out) {
163 int i;
164 sstr_append_of(out, "[", 1);
165 sstr_append_of_if(out, "\n", 1, indent);
166 curindent += indent;
167 for (i = 0; i < len; i++) {
168 sstr_append_indent(out, curindent);
169 sstr_printf_append(out, "%d", obj[i]);
170 if (i != len - 1) {
171 sstr_append_of(out, ",", 1);
172 }
173 sstr_append_of_if(out, "\n", 1, indent);
174 }
175 curindent -= indent;
176 sstr_append_indent(out, curindent);
177 sstr_append_of(out, "]", 1);
178 return 0;
179}
